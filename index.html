Function Overloading = Same name, different parameters.

#include <iostream>
using namespace std;

void print(int x) { cout << "Integer: " << x << endl; }

void print(string x) { cout << "String: " << x << endl; }

int main() {
    print(10);
    print(5.5);
    print("Hello!");
    return 0;
}
